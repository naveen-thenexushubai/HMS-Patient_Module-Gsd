@Component
public class PatientPermissionEvaluator implements PermissionEvaluator {

    @Autowired
    private PatientRepository patientRepository;

    @Override
    public boolean hasPermission(
        Authentication authentication,
        Object targetDomainObject,
        Object permission
    ) {
        if (authentication == null || !authentication.isAuthenticated()) {
            return false;
        }

        String targetType = targetDomainObject.getClass().getSimpleName();
        return hasPermission(authentication, targetDomainObject, targetType, permission.toString());
    }

    @Override
    public boolean hasPermission(
        Authentication authentication,
        Serializable targetId,
        String targetType,
        Object permission
    ) {
        if (authentication == null || !authentication.isAuthenticated()) {
            return false;
        }

        if (!"Patient".equals(targetType)) {
            return false;
        }

        String permissionString = permission.toString();
        String username = authentication.getName();

        // ADMIN has full access to all patients
        if (hasRole(authentication, "ADMIN")) {
            return true;
        }

        // RECEPTIONIST has full access for registration and administrative tasks
        if (hasRole(authentication, "RECEPTIONIST")) {
            return true;
        }

        // DOCTOR has read-only access to all patients
        // Phase 2 will add patient_assignments table check
        if (hasRole(authentication, "DOCTOR")) {
            return "read".equals(permissionString);
        }

        // NURSE has read-only access to all patients
        // Phase 2 will add care_team table check
        if (hasRole(authentication, "NURSE")) {
            return "read".equals(permissionString);
        }

        // Default deny
        return false;
    }

    private boolean hasRole(Authentication authentication, String role) {
        return authentication.getAuthorities().stream()
            .anyMatch(grantedAuthority ->
                grantedAuthority.getAuthority().equals("ROLE_" + role));
    }

    /**
     * Check if user can edit patient (used by frontend to show/hide Edit button)
     */
    public boolean canEdit(String username, String role) {
        return "ROLE_ADMIN".equals(role) || "ROLE_RECEPTIONIST".equals(role);
    }
}

@SpringBootTest
@ActiveProfiles("test")
@DirtiesContext(classMode = DirtiesContext.ClassMode.AFTER_CLASS)
public class PatientPermissionEvaluatorIntegrationTest {

    @Autowired
    private PatientPermissionEvaluator permissionEvaluator;

    @Autowired
    private PatientRepository patientRepository;

    private Patient testPatient;
    private UUID testBusinessId;

    @BeforeEach
    void setUp() {
        // Create test patient
        testPatient = Patient.builder()
            .firstName("Test")
            .lastName("Patient")
            .dateOfBirth(LocalDate.of(1990, 1, 1))
            .gender(Gender.MALE)
            .phoneNumber("555-123-4567")
            .status(PatientStatus.ACTIVE)
            .build();
        testPatient = patientRepository.save(testPatient);
        testBusinessId = testPatient.getBusinessId();
    }

    @Test
    void admin_hasFullAccess() {
        Authentication auth = createAuthentication("admin", "ROLE_ADMIN");

        assertTrue(permissionEvaluator.hasPermission(auth, testBusinessId, "Patient", "read"));
        assertTrue(permissionEvaluator.hasPermission(auth, testBusinessId, "Patient", "write"));
        assertTrue(permissionEvaluator.hasPermission(auth, testBusinessId, "Patient", "delete"));
    }

    @Test
    void receptionist_hasFullAccess() {
        Authentication auth = createAuthentication("receptionist", "ROLE_RECEPTIONIST");

        assertTrue(permissionEvaluator.hasPermission(auth, testBusinessId, "Patient", "read"));
        assertTrue(permissionEvaluator.hasPermission(auth, testBusinessId, "Patient", "write"));
        assertTrue(permissionEvaluator.hasPermission(auth, testBusinessId, "Patient", "delete"));
    }

    @Test
    void doctor_hasReadOnlyAccess() {
        Authentication auth = createAuthentication("doctor", "ROLE_DOCTOR");

        assertTrue(permissionEvaluator.hasPermission(auth, testBusinessId, "Patient", "read"));
        assertFalse(permissionEvaluator.hasPermission(auth, testBusinessId, "Patient", "write"));
        assertFalse(permissionEvaluator.hasPermission(auth, testBusinessId, "Patient", "delete"));
    }

    @Test
    void nurse_hasReadOnlyAccess() {
        Authentication auth = createAuthentication("nurse", "ROLE_NURSE");

        assertTrue(permissionEvaluator.hasPermission(auth, testBusinessId, "Patient", "read"));
        assertFalse(permissionEvaluator.hasPermission(auth, testBusinessId, "Patient", "write"));
        assertFalse(permissionEvaluator.hasPermission(auth, testBusinessId, "Patient", "delete"));
    }

    @Test
    void unauthenticated_hasNoAccess() {
        Authentication auth = null;

        assertFalse(permissionEvaluator.hasPermission(auth, testBusinessId, "Patient", "read"));
    }

    @Test
    void unknownRole_hasNoAccess() {
        Authentication auth = createAuthentication("guest", "ROLE_GUEST");

        assertFalse(permissionEvaluator.hasPermission(auth, testBusinessId, "Patient", "read"));
    }

    @Test
    void canEdit_returnsTrueForAdminAndReceptionist() {
        assertTrue(permissionEvaluator.canEdit("admin", "ROLE_ADMIN"));
        assertTrue(permissionEvaluator.canEdit("receptionist", "ROLE_RECEPTIONIST"));
        assertFalse(permissionEvaluator.canEdit("doctor", "ROLE_DOCTOR"));
        assertFalse(permissionEvaluator.canEdit("nurse", "ROLE_NURSE"));
    }

    private Authentication createAuthentication(String username, String... roles) {
        List<GrantedAuthority> authorities = Arrays.stream(roles)
            .map(SimpleGrantedAuthority::new)
            .collect(Collectors.toList());

        return new UsernamePasswordAuthenticationToken(username, "password", authorities);
    }
}

private List<EmergencyContactDto> mapEmergencyContacts(Patient patient) {
    List<EmergencyContact> contacts = emergencyContactRepository
        .findByPatientBusinessIdOrderByIsPrimaryDesc(patient.getBusinessId());

    return contacts.stream()
        .map(ec -> EmergencyContactDto.builder()
            .name(ec.getName())
            .phoneNumber(ec.getPhoneNumber())
            .relationship(ec.getRelationship())
            .isPrimary(ec.getIsPrimary())
            .build())
        .collect(Collectors.toList());
}

private MedicalHistoryDto mapMedicalHistory(Patient patient) {
    Optional<MedicalHistory> history = medicalHistoryRepository
        .findLatestByPatientBusinessId(patient.getBusinessId());

    return history.map(h -> MedicalHistoryDto.builder()
        .bloodGroup(h.getBloodGroup())
        .allergies(h.getAllergies())
        .chronicConditions(h.getChronicConditions())
        .build())
        .orElse(null);
}
