            .filter(p -> p.getPatientBusinessId().equals(patientBusinessId))
            .count();
        assertThat(totalPhotos).isEqualTo(2);
    }

    /**
     * Create minimal valid JPEG bytes for testing photo upload.
     * JPEG magic bytes: FF D8 FF E0 ... followed by minimal SOF and EOI markers.
     * This is the smallest valid JPEG that ImageIO.read() can parse.
     */
    private byte[] createMinimalJpeg() {
        // Minimal JPEG: SOI + APP0 + SOF0 + EOI — readable by ImageIO.read()
        // Using a small but valid JPEG structure that passes magic-byte validation
        return new byte[]{
            (byte) 0xFF, (byte) 0xD8,               // SOI (Start of Image)
            (byte) 0xFF, (byte) 0xE0,               // APP0 marker
            0x00, 0x10,                               // APP0 length (16 bytes)
            0x4A, 0x46, 0x49, 0x46, 0x00,           // "JFIF\0" identifier
            0x01, 0x01,                               // Version 1.1
            0x00,                                     // Aspect ratio units: no units
            0x00, 0x01, 0x00, 0x01,                  // X/Y density: 1x1
            0x00, 0x00,                               // Thumbnail width/height: none
            (byte) 0xFF, (byte) 0xDB,               // DQT (quantization table)
            0x00, 0x43, 0x00,                         // DQT length 67, table ID 0
            // 64 bytes of quantization values (all 16 = medium quality)
            0x10, 0x10, 0x10, 0x10, 0x10, 0x10, 0x10, 0x10,
            0x10, 0x10, 0x10, 0x10, 0x10, 0x10, 0x10, 0x10,
            0x10, 0x10, 0x10, 0x10, 0x10, 0x10, 0x10, 0x10,
            0x10, 0x10, 0x10, 0x10, 0x10, 0x10, 0x10, 0x10,
            0x10, 0x10, 0x10, 0x10, 0x10, 0x10, 0x10, 0x10,
            0x10, 0x10, 0x10, 0x10, 0x10, 0x10, 0x10, 0x10,
            0x10, 0x10, 0x10, 0x10, 0x10, 0x10, 0x10, 0x10,
            0x10, 0x10, 0x10, 0x10, 0x10, 0x10, 0x10, 0x10,
            (byte) 0xFF, (byte) 0xC0,               // SOF0 (Start of Frame)
            0x00, 0x0B,                               // SOF0 length (11 bytes)
            0x08,                                     // Precision: 8 bits
            0x00, 0x01, 0x00, 0x01,                  // Height: 1, Width: 1 (1x1 pixel)
            0x01,                                     // Components: 1 (grayscale)
            0x01, 0x11, 0x00,                         // Component 1: Y, 1x1 sampling, quant table 0
            (byte) 0xFF, (byte) 0xD9                // EOI (End of Image)
        };
    }
